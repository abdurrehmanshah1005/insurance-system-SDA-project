classDiagram
    direction TB

    class User {
        <<abstract>>
        -String id
        -String name
        -String contactNumber
        -String password
    }

    class Customer {
        -String registeredDate
        -List~Vehicle~ vehicles
        -List~Claim~ claimsHistory
    }

    class Staff {
        <<abstract>>
        -String employeeId
        -String role
    }

    class Vehicle {
        -String registrationNumber
        -String make
        -String model
        -String ownerId
        -InsurancePolicy policy
    }

    class InsurancePolicy {
        -String policyId
        -String issueDate
        -String expiryDate
        -Double coverageAmount
    }

    class Workshop {
        -String workshopId
        -String name
        -String location
        -Boolean isRegistered
    }

    class Claim {
        -String claimId
        -String dateFiled
        -String status
        -String description
        -String vehicleRegNum
        -String customerId
    }

    class InspectionReport {
        -String reportId
        -String claimId
        -String surveyorId
        -String workshopId
        -String damageDetails
        -Double estimatedCost
    }

    class ClaimManager {
        +fileNewClaim(customerId, vehicleReg, description)
        +assignSurveyor(claimId, surveyorId)
        +processApproval(claimId, managerId, decision)
    }

    class ReportGenerator {
        +generateNewCustomersReport(month, year)
        +generatePendingClaimsReport()
        +generateInspectionLogs()
    }

    class FileHandler {
        <<interface>>
        +saveRecord(filename, dataString)
        +loadAllRecords(filename)
        +findRecordById(filename, id)
        +updateRecord(filename, id, newData)
    }

    %% Inheritance
    User <|-- Customer
    User <|-- Staff

    %% Relationships and Multiplicity
    Customer "1" -- "*" Vehicle : owns
    Customer "1" -- "*" Claim : files
    Vehicle "1" -- "1" InsurancePolicy : covered by
    Claim "1" -- "1" InspectionReport : generates
    Staff "1" -- "*" InspectionReport : creates (Surveyor)
    InspectionReport "*" -- "1" Workshop : designates for repair
    
    %% Dependency on Persistence Layer
    ClaimManager ..> FileHandler : calls
    ReportGenerator ..> FileHandler : calls
