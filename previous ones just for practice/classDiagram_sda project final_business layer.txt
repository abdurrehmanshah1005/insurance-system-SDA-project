    %% ==========================================
    %% DOMAIN CLASSES (ENTITIES)
    %% ==========================================
    class Customer {
        -int customerId
        -string name
        -string contact
        -string address
        -Date registrationDate
        +display() void
    }

    class Vehicle {
        -int vehicleId
        -string registrationNo
        -string make
        -string model
        -int year
        -int customerId
        +display() void
    }

    class InsurancePolicy {
        -int policyId
        -int customerId
        -int vehicleId
        -Date startDate
        -Date endDate
        -double premium
        -PolicyStatus status
        +isActive(Date currentDate) bool
        +display() void
    }

    class Claim {
        -int claimId
        -int policyId
        -int vehicleId
        -Date accidentDate
        -Date claimDate
        -Date approvalDate
        -string damageDescription
        -ClaimStatus status
        -int surveyorId
        -int workshopId
        +updateStatus(ClaimStatus newStatus) void
        +display() void
    }

    class InspectionReport {
        -int inspectionId
        -int claimId
        -int surveyorId
        -Date inspectionDate
        -double estimatedCost
        -string recommendation
        -string remarks
        +display() void
    }

    class Workshop {
        -int workshopId
        -string name
        -string address
        -string contact
        -WorkshopStatus status
        +isActive() bool
        +display() void
    }

    class Staff {
        -int staffId
        -string name
        -string password
        -Role role
        +authenticate(string inputPassword) bool
        +display() void
    }

    %% ==========================================
    %% DATA ACCESS LAYER (REPOSITORIES)
    %% ==========================================
    class CustomerRepository {
        -string fileName
        +loadAll() vector~Customer~
        +saveAll(vector~Customer~ customers) void
        +findById(int id) optional~Customer~
        +add(Customer customer) void
        +update(Customer customer) bool
    }

    class VehicleRepository {
        -string fileName
        +loadAll() vector~Vehicle~
        +saveAll(vector~Vehicle~ vehicles) void
        +findById(int id) optional~Vehicle~
        +findByCustomer(int customerId) vector~Vehicle~
        +add(Vehicle vehicle) void
        +update(Vehicle vehicle) bool
    }

    class PolicyRepository {
        -string fileName
        +loadAll() vector~InsurancePolicy~
        +saveAll(vector~InsurancePolicy~ policies) void
        +findById(int id) optional~InsurancePolicy~
        +findByVehicle(int vehicleId) optional~InsurancePolicy~
        +findByCustomer(int customerId) vector~InsurancePolicy~
        +add(InsurancePolicy policy) void
        +update(InsurancePolicy policy) bool
    }

    class ClaimRepository {
        -string fileName
        +loadAll() vector~Claim~
        +saveAll(vector~Claim~ claims) void
        +findById(int id) optional~Claim~
        +findByPolicy(int policyId) vector~Claim~
        +findByCustomer(int customerId) vector~Claim~
        +findPending() vector~Claim~
        +add(Claim claim) void
        +update(Claim claim) bool
    }

    class InspectionRepository {
        -string fileName
        +loadAll() vector~InspectionReport~
        +saveAll(vector~InspectionReport~ reports) void
        +findByClaimId(int claimId) optional~InspectionReport~
        +add(InspectionReport report) void
        +update(InspectionReport report) bool
    }

    class WorkshopRepository {
        -string fileName
        +loadAll() vector~Workshop~
        +saveAll(vector~Workshop~ workshops) void
        +findById(int id) optional~Workshop~
        +getActiveWorkshops() vector~Workshop~
        +add(Workshop workshop) void
        +update(Workshop workshop) bool
    }

    class StaffRepository {
        -string fileName
        +loadAll() vector~Staff~
        +saveAll(vector~Staff~ staffList) void
        +findById(int id) optional~Staff~
        +authenticate(int id, string password) optional~Staff~
        +add(Staff staff) void
        +update(Staff staff) bool
    }

    %% ==========================================
    %% BUSINESS LOGIC LAYER (SERVICES)
    %% ==========================================
    class CustomerService {
        +registerCustomer(string name, string contact, string address) Customer
        +getCustomer(int id) optional~Customer~
    }

    class VehicleService {
        +addVehicle(int custId, string regNo, string make, string model, int year) Vehicle
        +getVehiclesByCustomer(int customerId) vector~Vehicle~
    }

    class PolicyService {
        +createPolicy(int custId, int vehId, Date start, Date end, double premium) InsurancePolicy
        +getPolicy(int policyId) optional~InsurancePolicy~
        +isPolicyActive(int policyId) bool
    }

    class ClaimService {
        +createClaim(int polId, int vehId, Date accidentDate, string desc) Claim
        +assignSurveyor(int claimId, int surveyorId) bool
        +approveClaim(int claimId) bool
        +rejectClaim(int claimId) bool
    }

    class InspectionService {
        +createInspection(int claimId, int survId, double cost, string rec, string remarks) InspectionReport
        +getInspectionByClaim(int claimId) optional~InspectionReport~
    }

    class WorkshopService {
        +assignWorkshopToClaim(int claimId, int workshopId) bool
        +getActiveWorkshops() vector~Workshop~
    }

    class ReportService {
        +getNewCustomersByMonth(int month, int year) vector~Customer~
        +getPendingClaims() vector~Claim~
        +getCustomerClaimHistory(int customerId) vector~Claim~
    }

    class AuthService {
        +login(int staffId, string password) optional~Staff~
    }

    %% ==========================================
    %% PRESENTATION LAYER (UI)
    %% ==========================================
    class InsuranceApp {
        +run() void
        -showMainMenu() void
        -salesmanMenu() void
        -surveyorMenu() void
        -managerMenu() void
    }

    %% ==========================================
    %% RELATIONSHIPS & MULTIPLICITIES
    %% ==========================================

    %% Domain Relationships
    Customer "1" --> "*" Vehicle : owns
    Vehicle "1" --> "0..1" InsurancePolicy : has
    InsurancePolicy "1" --> "*" Claim : generates
    Claim "1" --> "0..1" InspectionReport : has
    Workshop "1" --> "0..*" Claim : repairs
    Staff "1" --> "*" InspectionReport : creates
    Staff "1" --> "*" Claim : approves/reviews

    %% UI to Service Dependencies
    InsuranceApp ..> AuthService : uses
    InsuranceApp ..> CustomerService : uses
    InsuranceApp ..> VehicleService : uses
    InsuranceApp ..> PolicyService : uses
    InsuranceApp ..> ClaimService : uses
    InsuranceApp ..> InspectionService : uses
    InsuranceApp ..> WorkshopService : uses
    InsuranceApp ..> ReportService : uses

    %% Service to Repository Dependencies
    CustomerService ..> CustomerRepository : uses
    VehicleService ..> VehicleRepository : uses
    PolicyService ..> PolicyRepository : uses
    ClaimService ..> ClaimRepository : uses
    ClaimService ..> PolicyRepository : uses
    ClaimService ..> InspectionRepository : uses
    InspectionService ..> InspectionRepository : uses
    WorkshopService ..> WorkshopRepository : uses
    AuthService ..> StaffRepository : uses
    ReportService ..> CustomerRepository : uses
    ReportService ..> ClaimRepository : uses
    ReportService ..> InspectionRepository : uses
